81  void loop() {
82  keepAlive();
83  if (getReferenceClock() == nullptr) return;
84 
85  unsigned long nowMillis = clockMillis();
86 
87  // Finite state machine based on mRequestStatus
88  switch (mRequestStatus) {
89  case kStatusReady:
90  getReferenceClock()->sendRequest();
91  mRequestStartMillis = nowMillis;
92  mRequestStatus = kStatusSent;
93  setSecondsSinceSyncAttempt(0);
94  setSecondsToSyncAttempt(mCurrentSyncPeriodSeconds);
95  break;
96 
97  case kStatusSent: {
98  unsigned long elapsedMillis = nowMillis - mRequestStartMillis;
99  setSecondsSinceSyncAttempt(elapsedMillis / 1000);
100  if (mTimingStats) mTimingStats->update(elapsedMillis);
101 
102  if (getReferenceClock()->isResponseReady()) {
103  acetime_t nowSeconds = getReferenceClock()->readResponse();
104 
105  // If response came back but was invalid, reschedule.
106  if (nowSeconds == kInvalidSeconds) {
107  mRequestStatus = kStatusWaitForRetry;
108  setSyncStatusCode(kSyncStatusError);
109  } else {
110  syncNow(nowSeconds);
111  mCurrentSyncPeriodSeconds = mSyncPeriodSeconds;
112  mLastSyncMillis = nowMillis;
113  mRequestStatus = kStatusOk;
114  setSyncStatusCode(kSyncStatusOk);
115  }
116  } else {
117  // If timed out, reschedule.
118  if (elapsedMillis >= mRequestTimeoutMillis) {
119  mRequestStatus = kStatusWaitForRetry;
120  setSyncStatusCode(kSyncStatusTimedOut);
121  }
122  }
123  setSecondsToSyncAttempt(mCurrentSyncPeriodSeconds);
124  break;
125  }
126 
127  // The previous request succeeded, so wait until the next sync attempt.
128  case kStatusOk: {
129  unsigned long elapsedMillis = nowMillis - mLastSyncMillis;
130  setSecondsSinceSyncAttempt(elapsedMillis / 1000);
131  if (elapsedMillis >= mCurrentSyncPeriodSeconds * 1000UL) {
132  mRequestStatus = kStatusReady;
133  }
134  setSecondsToSyncAttempt((int32_t) mCurrentSyncPeriodSeconds
135  - getSecondsSinceSyncAttempt());
136  break;
137  }
138 
139  // Previous request failed, so update timing parameters so that
140  // subsequent loop() retries with an exponential backoff, until a
141  // maximum of mSyncPeriodSeconds is reached.
142  case kStatusWaitForRetry: {
143  unsigned long elapsedMillis = nowMillis - mRequestStartMillis;
144  setSecondsSinceSyncAttempt(elapsedMillis / 1000);
145 
146  // Adjust mCurrentSyncPeriodSeconds using exponential backoff.
147  if (elapsedMillis >= mCurrentSyncPeriodSeconds * 1000UL) {
148  if (mCurrentSyncPeriodSeconds >= mSyncPeriodSeconds / 2) {
149  mCurrentSyncPeriodSeconds = mSyncPeriodSeconds;
150  } else {
151  mCurrentSyncPeriodSeconds *= 2;
152  }
153  mRequestStatus = kStatusReady;
154  setSecondsToSyncAttempt(0);
155  } else {
156  setSecondsToSyncAttempt((int32_t) mCurrentSyncPeriodSeconds
157  - getSecondsSinceSyncAttempt());
158  }
159  break;
160  }
161  }
162  }
