434 inline bool operator==(const TimeZone& a, const TimeZone& b) {
435  if (a.mType != b.mType) return false;
436  switch (a.mType) {

441  && a.mDstOffsetCode == b.mDstOffsetCode;

443  case TimeZone::kTypeExtended:
444  case TimeZone::kTypeBasicManaged:

447  default:
