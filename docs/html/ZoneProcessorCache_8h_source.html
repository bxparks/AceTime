73  ZP* findUsingZoneKey(uintptr_t zoneKey) {
74  for (uint8_t i = 0; i < mSize; i++) {
75  ZP* zoneProcessor = &mZoneProcessors[i];
76  if (zoneProcessor->equalsZoneKey(zoneKey)) {
77  return zoneProcessor;
78  }
