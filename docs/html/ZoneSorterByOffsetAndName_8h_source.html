90  if (a.isNull()) {
91  if (b.isNull()) {

99  int16_t offsetA = a.stdOffsetMinutes();
100  int16_t offsetB = b.stdOffsetMinutes();
101  if (offsetA < offsetB) return -1;
102  if (offsetA > offsetB) return 1;
