130  static bool isSorted(const ZI* const* registry, uint16_t registrySize) {
131  if (registrySize == 0) {
132  return false;
133  }
134 
135  const ZRB zoneRegistry(registry);
136  uint32_t prevId = ZIB(zoneRegistry.zoneInfo(0)).zoneId();
137  for (uint16_t i = 1; i < registrySize; ++i) {
138  uint32_t currentId = ZIB(zoneRegistry.zoneInfo(i)).zoneId();
139  if (prevId > currentId) {
140  return false;
141  }
142  prevId = currentId;
143  }
144  return true;
145  }

151  static uint16_t linearSearchById(const ZI* const* registry,
152  uint16_t registrySize, uint32_t zoneId) {
153  const ZRB zoneRegistry(registry);
154  for (uint16_t i = 0; i < registrySize; ++i) {
155  const ZI* zoneInfo = zoneRegistry.zoneInfo(i);

183  static uint16_t binarySearchById(const ZI* const* registry,
184  uint16_t registrySize, uint32_t zoneId) {
185  uint16_t a = 0;
186  uint16_t b = registrySize;
187  const ZRB zoneRegistry(registry);
188  while (true){
189  uint16_t diff = b - a;
190  if (diff == 0) break;
191 
192  uint16_t c = a + diff / 2;
193  const ZI* zoneInfo = zoneRegistry.zoneInfo(c);
194  uint32_t currentId = ZIB(zoneInfo).zoneId();
195  if (currentId == zoneId) return c;
196  if (zoneId < currentId) {
197  b = c;
198  } else {
199  a = c + 1;
200  }
201  }
202  return kInvalidIndex;
203  }
