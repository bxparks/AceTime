65  ZP* findUsingZoneKey(uintptr_t zoneKey) {
66  for (uint8_t i = 0; i < SIZE; i++) {
67  ZP* zoneProcessor = &mZoneProcessors[i];
68  if (zoneProcessor->equalsZoneKey(zoneKey)) {
69  return zoneProcessor;
70  }
