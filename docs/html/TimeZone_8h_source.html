366  OffsetDateTime getOffsetDateTime(const LocalDateTime& ldt) const {
367  OffsetDateTime odt = OffsetDateTime::forError();
368  switch (mType) {
369  case kTypeError:
370  case kTypeReserved:
371  break;
372 
373  case kTypeManual:
374  odt = OffsetDateTime::forLocalDateTimeAndOffset(
375  ldt,
376  TimeOffset::forMinutes(mStdOffsetMinutes + mDstOffsetMinutes));
377  break;
378 
379  default: {
380  FindResult result = getBoundZoneProcessor()->findByLocalDateTime(ldt);
381  if (result.type == FindResult::kTypeNotFound) {
382  break;
383  }
384 
385  // Convert FindResult into OffsetDateTime using the requested offset.
386  TimeOffset reqOffset = TimeOffset::forMinutes(
387  result.reqStdOffsetMinutes + result.reqDstOffsetMinutes);
388  odt = OffsetDateTime::forLocalDateTimeAndOffset(ldt, reqOffset);
389  odt.fold(result.fold);
390 
391  // Special processing for kTypeGap: Convert to epochSeconds using the
392  // reqStdOffsetMinutes and reqDstOffsetMinutes, then convert back to
393  // OffsetDateTime using the target stdOffsetMinutes and
394  // dstOffsetMinutes.
395  if (result.type == FindResult::kTypeGap) {
396  acetime_t epochSeconds = odt.toEpochSeconds();
397  TimeOffset targetOffset = TimeOffset::forMinutes(
398  result.stdOffsetMinutes + result.dstOffsetMinutes);
399  odt = OffsetDateTime::forEpochSeconds(epochSeconds, targetOffset);
400  }
401  break;
402  }
403  }
404  return odt;
405  }
