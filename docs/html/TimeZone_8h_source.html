532 inline bool operator==(const TimeZone& a, const TimeZone& b) {
533  if (a.mType != b.mType) return false;
534  switch (a.mType) {
535  case TimeZone::kTypeError:
536  return true;
537  case TimeZone::kTypeManual:
538  return a.mStdOffsetMinutes == b.mStdOffsetMinutes
539  && a.mDstOffsetMinutes == b.mDstOffsetMinutes;
540  case TimeZone::kTypeBasic:
541  case TimeZone::kTypeExtended:
542  case TimeZone::kTypeBasicManaged:
543  case TimeZone::kTypeExtendedManaged:
544  return (a.mZoneInfo == b.mZoneInfo);
545  default:
546  return false;
547  }
