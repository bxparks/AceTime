142  return createForZoneInfo(zoneInfo);

147  return createForZoneInfo(zoneInfo);
148  }
149 
150  TimeZone createForZoneIndex(uint16_t index) override {
151  const ZI* zoneInfo = mZoneRegistrar.getZoneInfoForIndex(index);
152  return createForZoneInfo(zoneInfo);
153  }
154 
155  TimeZone createForTimeZoneData(const TimeZoneData& d) override {
156  switch (d.type) {
157  case TimeZoneData::kTypeError:
158  return TimeZone::forError();
159  case TimeZoneData::kTypeManual:
160  return TimeZone::forTimeOffset(
161  TimeOffset::forMinutes(d.stdOffsetMinutes),
162  TimeOffset::forMinutes(d.dstOffsetMinutes));
163  case TimeZoneData::kTypeZoneId:
164  return createForZoneId(d.zoneId);
165  default:
166  // Maybe this should return TimeZone::forError()?
167  return TimeZone();
168  }
169  }
170 
171  uint16_t indexForZoneName(const char* name) const override {
172  return mZoneRegistrar.findIndexForName(name);
173  }
174 
175  uint16_t indexForZoneId(uint32_t id) const override {
176  return mZoneRegistrar.findIndexForId(id);
177  }
178 
179  uint16_t registrySize() const override {
180  return mZoneRegistrar.registrySize();
181  }
182 
189  TimeZone createForZoneInfo(const ZI* zoneInfo) {
190  if (! zoneInfo) return TimeZone::forError();
191  return TimeZone(zoneInfo, &mZoneProcessorCache);
192  }
193 
194  protected:
201  ZoneManagerImpl(uint16_t registrySize, const ZI* const* zoneRegistry):
202  mZoneRegistrar(registrySize, zoneRegistry),
203  mZoneProcessorCache() {}
