70  acetime_t getNow() const override {
71  if (!mIsInit) return kInvalidSeconds;
72 
73  // Update mEpochSeconds by the number of seconds elapsed according to the
74  // millis(). This method is expected to be called multiple times a second,
75  // so the while() loop below will normally execute 0 times, until the
76  // millis() clock goes past the mPrevKeepAliveMillis by 1 second.
77  //
78  // There are 2 reasons why this method will be called multiple times a
79  // second:
80  //
81  // 1) A physical clock with an external display will want to refresh
82  // its display 5-10 times a second, so that it can capture the transition
83  // from one second to the next without too much jitter. So it will call
84  // this method multiple times a second to check if one second has passed.
85  //
86  // 2) If the SystemClockCoroutine or SystemClockLoop classes is used,
87  // then the keepAlive() method will be called perhaps 100's times per
88  // second, as fast as the iteration speed of the global loop() function.
89  while ((uint16_t) ((uint16_t) clockMillis() - mPrevKeepAliveMillis)
90  >= 1000) {
91  mPrevKeepAliveMillis += 1000;
92  mEpochSeconds += 1;
93  }
94 
95  return mEpochSeconds;
96  }

288  void syncNow(acetime_t epochSeconds) {
289  if (epochSeconds == kInvalidSeconds) return;
290 
291  mLastSyncTime = epochSeconds;
292  acetime_t skew = mEpochSeconds - epochSeconds;
